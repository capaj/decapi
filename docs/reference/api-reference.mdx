# Api Reference

## @ObjectType

Registers a class as a GraphQL object type.

```ts
interface ObjectTypeOptions {
  name?: string; // infered from class name
  description?: string;
}

@ObjectType(options?: ObjectTypeOptions)
```

## @Field

Registers a class field as a GraphQLField.

```ts
interface IFieldOptions {
  description?: string
  type?: any
  castTo?: any
  onlyDecoratedArgs?: boolean // when true this will not expose arguments that aren't decorated by @Arg()
  // useful when you have optional arguments that you only want to use on the backend
  name?: string
  nullable?: boolean
}

@Field(options?: FieldOptions)
```

## @Before

```ts

  source: any;
  args: { [argName: string]: any };
  context: any;
  info: GraphQLResolveInfo;
}

type HookExecutor = (data: HookExecutorResolverArgs) => any | Promise<any>;

@Before(hook: HookExecutor);
```

## @SchemaRoot

```ts
@SchemaRoot(): ClassDecorator;
```

### @Query

Has same interface as [@Field](#field) decorator. Can be used only inside @SchemaRoot class

### @Mutation

Has same interface as [@Field](#field) decorator. Can be used only inside @SchemaRoot class

### @QueryAndMutation

A shorthand for both Query And Mutation. Can be used only inside @SchemaRoot class

## @Union

```ts
interface UnionOptions {
  name?: string;
  resolveTypes?: (value: any, context: any, info: GraphQLResolveInfo): any; // must return type resolvable to one of defined in `types` option
  types: any[] | () => any[];
}

interface UnionTypeResolver {
  (value: any, context: any, info: GraphQLResolveInfo): any;
}

@Union(options: UnionOptions): ClassDecorator;
```
